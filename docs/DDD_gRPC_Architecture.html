6.2 gRPC Server Implementation

OrderGrpcService.java Java 
@GrpcService
public class OrderGrpcService extends OrderServiceGrpc.OrderServiceImplBase {

    private final OrderApplicationService orderService;
    private final OrderProtoMapper mapper;

    @Override
    public void createOrder(
            CreateOrderRequest request,
            StreamObserver<OrderResponse> responseObserver) {
        
        try {
            // Convert Protobuf ‚Üí Command
            CreateOrderCommand command = mapper.toCommand(request);
            
            // Execute use case
            OrderDto result = orderService.createOrder(command);
            
            // Convert Domain ‚Üí Protobuf Response
            OrderResponse response = mapper.toProto(result);
            
            responseObserver.onNext(response);
            responseObserver.onCompleted();
            
        } catch (DomainException e) {
            responseObserver.onError(
                Status.INVALID_ARGUMENT
                    .withDescription(e.getMessage())
                    .asRuntimeException()
            );
        }
    }

    @Override
    public void getOrderUpdates(
            OrderIdRequest request,
            StreamObserver<OrderResponse> responseObserver) {
        
        // Server streaming: push updates as they happen
        orderService.subscribeToOrderUpdates(request.getOrderId())
            .subscribe(
                order -> responseObserver.onNext(mapper.toProto(order)),
                error -> responseObserver.onError(Status.INTERNAL.asRuntimeException()),
                () -> responseObserver.onCompleted()
            );
    }
}

            new CustomerName(proto.getFirstName(), proto.getLastName()),

            toAddress(proto.getAddress())  // Convert nested
