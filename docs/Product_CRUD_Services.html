üíª Create Implementation

@Transactional
public ProductDetailResponse create(ProductRequest request) {
    // 1. Generate unique key
    String key = UUID.randomUUID().toString();
    
    // 2. Generate slug
    String slug = generateSlug(request.getSlug(), request.getName());
    
    // 3. Normalize category
    String category = normalizeCategory(request.getCategory());
    
    // 4. Build entity
    ProductEntity entity = ProductEntity.builder()
        .key(key)
        .name(request.getName())
        .slug(slug)
        .category(category)
        .price(request.getPrice())
        .gallery(convertGalleryToString(request.getGallery()))
        .materials(convertListToJson(request.getMaterials()))
        .colors(convertListToJson(request.getColors()))
        .isDeleted(false)
        .build();
    
    // 5. Save
    ProductEntity saved = productRepository.save(entity);
    
    // 6. Return response
    return toDetailResponse(saved, false, List.of());
}

üîç List with Filtering

public Page list(
    ProductFilterRequest filter, 
    Pageable pageable, 
    Long currentUserId) {
    
    // 1. Get favorite keys for current user
    Set favoriteKeys = currentUserId != null
        ? favoriteRepository.findByUserId(currentUserId)
            .stream()
            .map(FavoriteEntity::getProductKey)
            .collect(Collectors.toSet())
        : Set.of();
    
    // 2. Query with filters
    Page page = productRepository.findAll(
        ProductSpecification.withFilters(filter), 
        pageable
    );
    
    // 3. Map to response
    return page.map(entity -> 
        toListItemResponse(entity, favoriteKeys)
    );
}
